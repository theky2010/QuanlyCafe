function cau1()
{

}

function cau2()
{
echo  "Ten tap tin muon doi"

function cau3()
{
echo -n "n= "
read n
for ((i=0; i<n; i++))
do
echo -n "a[$i] " =
read a[$i]
done
}
function max2nums()
{
if [ "$1" -gt "$2" ]; then
max1=$1
echo $1
else
max1=$2
echo $2
fi
return $max1
}
function min2nums()
{
if [ "$1" -lt "$2" ]; then
min1=$1
echo $1
else
min1=$2
echo $2
fi
return $min1
}
#max of an array
function maxs()
{
max=${a[0]}
for ((i=1; i<n; i++))
do
max=$(max2nums ${a[$i]} $max)
done
echo "max = $max"
}
#min of an array
function mins()
{
min=${a[0]}
for ((i=1; i<n; i++))
do
min=$(min2nums ${a[$i]} $min)
done
echo "min = $min"
}

function cau4()
{
echo "Nhap so"
read n
i=2
while [ $i -lt $n ]
do
sum=$((sum+i))
i=$((i+2))
done
echo "tong cac so chan cua $n la: "$sum
}

function cau5snt()
{
x=$1
n=2
while [ $n -le $x ]
do
i=2
count=1
#kiem tra snt
while [ $i -lt $n ]
do
if [ `expr $n % $i` -eq 0 ]
then
count=0
break
fi
i=`expr $i + 1`
done
#ket thuc kiem tra
if [ $count -eq 1 ]
then
echo "$n la so nguyen to"
fi

n=`expr $n + 1`
done
}
#in ra so chinh phuong
function cau5sochinhphuong()
{
x=$1
n=2
while [ $n -le $x ]
do
i=2
count=1
sochinhphuong $n
n=`expr $n + 1`
done
}
#kiem tra so chinh phuong
function sochinhphuong()
{
num=$1
for (( i = 1; i*i <= num; i++ ))
do
	if test `expr $i \* $i` -eq $num ; then
echo "$num la so chinh phuong"
return
fi
done
}
